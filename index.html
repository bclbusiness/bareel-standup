// ── PARTICIPANT JOIN ────────────────────────────────────────────────────────

function buildCodeInputs() {

  const wrap = document.getElementById('code-input-boxes');

  wrap.innerHTML = '';

  for (let i = 0; i < 6; i++) {

    const inp = document.createElement('input');

    inp.type = 'text';

    inp.maxLength = 1;

    inp.className = 'code-input-box';

    inp.inputMode = 'numeric';

    inp.pattern = '[0-9]*';

    inp.dataset.index = i;

    inp.addEventListener('input', e => {

      e.target.value = e.target.value.replace(/\D/g,'');

      if (e.target.value && i < 5) wrap.children[i+1].focus();

    });

    inp.addEventListener('keydown', e => {

      if (e.key === 'Backspace' && !e.target.value && i > 0) wrap.children[i-1].focus();

    });

    inp.addEventListener('paste', e => {

      e.preventDefault();

      const text = (e.clipboardData || window.clipboardData).getData('text').replace(/\D/g,'').slice(0,6);

      text.split('').forEach((ch, idx) => { if (wrap.children[idx]) wrap.children[idx].value = ch; });

      const next = Math.min(text.length, 5);

      wrap.children[next].focus();

    });

    wrap.appendChild(inp);

  }

}


async function joinRoom() {

  const btn   = document.getElementById('btn-join');

  const errEl = document.getElementById('join-error');

  errEl.classList.remove('visible');


  if (!db) { showError('join-error', 'Supabase is not configured yet. Please contact the standup organiser.'); return; }


  const boxes = document.querySelectorAll('.code-input-box');

  const code  = Array.from(boxes).map(b => b.value).join('');

  if (code.length < 6) { showError('join-error', 'Please enter the full 6-digit code.'); return; }


  const name = document.getElementById('participant-name').value.trim();

  if (!name) { showError('join-error', 'Please enter your name.'); return; }


  btn.innerHTML = '<div class="spinner"></div> Joining...';

  btn.disabled = true;


  const { data: room, error } = await db.from('rooms').select('*').eq('code', code).single();


  if (error || !room) {

    btn.innerHTML = 'Join standup'; btn.disabled = false;

    showError('join-error', 'Room not found. Check the code and try again.'); return;

  }

  if (room.status === 'complete') {

    btn.innerHTML = 'Join standup'; btn.disabled = false;

    showError('join-error', 'This standup has already finished.'); return;

  }


  const updatedParticipants = [...(room.participants || [])];

  if (!updatedParticipants.includes(name)) updatedParticipants.push(name);


  await db.from('rooms').update({ participants: updatedParticipants }).eq('id', room.id);


  state.role = 'participant';

  state.roomId = room.id;

  state.roomCode = code;

  state.participantName = name;

  state.questions = room.questions;

  state.totalSeconds = room.total_seconds;


  document.getElementById('join-confirmed-name').textContent = name;

  document.getElementById('join-waiting-name-label').textContent = `Waiting for the admin to begin — ${room.name}`;


  document.getElementById('join-form-wrap').classList.add('hidden');

  document.getElementById('join-waiting-wrap').classList.remove('hidden');


  btn.innerHTML = 'Join standup'; btn.disabled = false;


  subscribeToRoom(room.id);


  // if standup already active, jump straight in

  if (room.status === 'active') renderActiveScreen(room);

  if (room.status === 'complete') showScreen('screen-complete');

}


// ── BEGIN STANDUP ──────────────────────────────────────────────────────────

async function beginStandup() {

  const participants = state.participants;

  if (participants.length === 0) return;


  const shuffled = [...participants].sort(() => Math.random() - 0.5);

  const perPersonSeconds = Math.floor(state.totalSeconds / participants.length);


  await db.from('rooms').update({

    status: 'active',

    speaker_order: shuffled,

    current_index: 0,

    is_paused: false,

    turn_started_at: new Date().toISOString(),

    per_person_seconds: perPersonSeconds,

  }).eq('id', state.roomId);

}


// ── ACTIVE STANDUP ─────────────────────────────────────────────────────────

function renderActiveScreen(room) {

  state.speakerOrder     = room.speaker_order || [];

  state.currentIndex     = room.current_index || 0;

  state.isPaused         = room.is_paused || false;

  state.perPersonSeconds = room.per_person_seconds || Math.floor(state.totalSeconds / state.participants.length);

  state.questions        = room.questions || state.questions;

  state.participants     = room.participants || state.participants;


  if (state.currentIndex >= state.speakerOrder.length) {

    showCompleteScreen(room); return;

  }


  showScreen('screen-active');


  // admin controls visibility

  document.getElementById('admin-controls').style.display = state.role === 'admin' ? 'block' : 'none';


  updateActiveSpeaker(room);

  startLocalTimer(room);

}


function updateActiveSpeaker(room) {

  const speaker = state.speakerOrder[state.currentIndex];

  const ini = initials(speaker);


  document.getElementById('active-avatar').textContent = ini;

  document.getElementById('active-speaker-name').textContent = speaker;


  const perSec = state.perPersonSeconds;

  const m = Math.floor(perSec/60), s = perSec%60;

  document.getElementById('active-speaker-alloc').textContent =

    s > 0 ? `${m} min ${s} sec allocated` : `${m} min allocated`;


  // questions

  const qEl = document.getElementById('active-questions');

  qEl.innerHTML = state.questions.map((q,i) => `

    <div class="question-item">

      <div class="q-num">${i+1}</div>

      <div class="q-text">${q}</div>

    </div>

  `).join('');


  // remaining

  const remaining = state.speakerOrder.slice(state.currentIndex + 1);

  const pillsEl = document.getElementById('remaining-pills');

  if (remaining.length === 0) {

    pillsEl.innerHTML = '<span style="font-size:13px;color:var(--text-tertiary)">No one — this is the last speaker!</span>';

  } else {

    pillsEl.innerHTML = remaining.map(p => `

      <span class="remaining-pill">

        <span class="pill-avatar">${initials(p)}</span>${p}

      </span>

    `).join('');

  }


  // progress

  const done = state.currentIndex;

  const total = state.speakerOrder.length;

  const pct = (done / total) * 100;

  document.getElementById('progress-bar').style.width = pct + '%';

  document.getElementById('topbar-progress-label').textContent = `${done} of ${total} done`;


  // paused banner

  document.getElementById('paused-banner').classList.toggle('visible', state.isPaused);

  const pauseLabel = document.getElementById('pause-label');

  if (pauseLabel) pauseLabel.textContent = state.isPaused ? 'Resume timer' : 'Pause timer';

}


function startLocalTimer(room) {

  clearInterval(state.timerInterval);


  const turnStarted = new Date(room.turn_started_at).getTime();

  const totalMs     = state.perPersonSeconds * 1000;

  const circumference = 452;


  function tick() {

    if (state.isPaused) return;

    const elapsed = Date.now() - turnStarted;

    const remaining = Math.max(0, totalMs - elapsed);

    state.timeLeft = Math.ceil(remaining / 1000);


    // update timer display

    const m = Math.floor(state.timeLeft / 60);

    const s = state.timeLeft % 60;

    document.getElementById('timer-text').textContent = `${m}:${String(s).padStart(2,'0')}`;


    const progress = remaining / totalMs;

    const offset   = circumference * (1 - progress);

    document.getElementById('timer-circle').style.strokeDashoffset = offset;


    // turn red when ≤ 30s

    document.getElementById('timer-circle').style.stroke =

      state.timeLeft <= 30 ? '#C0392B' : 'var(--brand-mid)';

    document.getElementById('timer-text').style.fill =

      state.timeLeft <= 30 ? '#C0392B' : '#26215C';


    if (remaining <= 0) {

      clearInterval(state.timerInterval);

      if (state.role === 'admin') advanceToNext();

    }

  }


  tick();

  state.timerInterval = setInterval(tick, 500);

}


// ── NEXT PERSON ────────────────────────────────────────────────────────────

async function nextPerson() {

  clearInterval(state.timerInterval);

  await advanceToNext();

}


async function advanceToNext() {

  const nextIndex = state.currentIndex + 1;

  if (nextIndex >= state.speakerOrder.length) {

    await db.from('rooms').update({ status: 'complete', current_index: nextIndex }).eq('id', state.roomId);

    return;

  }

  await db.from('rooms').update({

    current_index: nextIndex,

    turn_started_at: new Date().toISOString(),

    is_paused: false,

  }).eq('id', state.roomId);

}


// ── PAUSE ──────────────────────────────────────────────────────────────────

async function togglePause() {

  const newPaused = !state.isPaused;

  await db.from('rooms').update({ is_paused: newPaused }).eq('id', state.roomId);

}


// ── COMPLETE ───────────────────────────────────────────────────────────────

function showCompleteScreen(room) {

  clearInterval(state.timerInterval);

  const participants = room.participants || state.participants;

  const totalMins = Math.round(state.totalSeconds / 60);

  const perSec = state.perPersonSeconds || Math.floor(state.totalSeconds / participants.length);

  const perM = Math.floor(perSec/60), perS = perSec%60;


  document.getElementById('complete-count').textContent = participants.length;

  document.getElementById('complete-total').textContent = `${totalMins}m`;

  document.getElementById('complete-per').textContent = perS > 0 ? `${perM}m ${perS}s` : `${perM}m`;


  document.getElementById('complete-participants').innerHTML = participants.map(p => `

    <div class="remaining-pill">

      <span class="pill-avatar">${initials(p)}</span>${p}

    </div>

  `).join('');


  showScreen('screen-complete');

}


function runAnother() {

  clearInterval(state.timerInterval);

  if (state.subscription) state.subscription.unsubscribe();

  state = { role: null, roomId: null, roomCode: null, participantName: null,

    participants: [], questions: [], totalSeconds: 0, perPersonSeconds: 0,

    speakerOrder: [], currentIndex: 0, timeLeft: 0, isPaused: false,

    timerInterval: null, subscription: null };


  if (window.location.search.includes('join')) {

    showScreen('screen-join');

  } else {

    showScreen('screen-setup');

  }

}


// ── REALTIME SUBSCRIPTION ──────────────────────────────────────────────────

function subscribeToRoom(roomId) {

  if (state.subscription) state.subscription.unsubscribe();


  state.subscription = db

    .channel(`room-${roomId}`)

    .on('postgres_changes', {

      event: 'UPDATE', schema: 'public', table: 'rooms', filter: `id=eq.${roomId}`

    }, payload => {

      const room = payload.new;


      if (room.status === 'waiting') {

        if (state.role === 'admin') updateWaitingParticipants(room.participants || []);

      }


      if (room.status === 'active') {

        state.currentIndex     = room.current_index;

        state.isPaused         = room.is_paused;

        state.perPersonSeconds = room.per_person_seconds;

        state.speakerOrder     = room.speaker_order;

        state.participants     = room.participants;


        if (document.getElementById('screen-active').classList.contains('active')) {

          updateActiveSpeaker(room);

          startLocalTimer(room);

        } else {

          renderActiveScreen(room);

        }

      }


      if (room.status === 'complete') {

        showCompleteScreen(room);

      }

    })

    .subscribe();

}


// ── UTILITIES ──────────────────────────────────────────────────────────────

function initials(name) {

  if (!name) return '?';

  const parts = name.trim().split(' ').filter(Boolean);

  if (parts.length === 1) return parts[0][0].toUpperCase();

  return (parts[0][0] + parts[parts.length-1][0]).toUpperCase();

}


function showError(id, msg) {

  const el = document.getElementById(id);

  el.textContent = msg;

  el.classList.add('visible');

}
